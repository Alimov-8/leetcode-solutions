# Leetcode 206. Reverse the LinkedList 



## Recursion approach 
```py
# Time Complexity - 

# Definition for singly-linked list.
# class ListNode:
#     def __init__(self, val=0, next=None):
#         self.val = val
#         self.next = next
class Solution:
    def reverseList(self, head: Optional[ListNode]) -> Optional[ListNode]:
        def reverse(head):
            
            node = None
            
            # Terminate if empty LinkedList is given
            if head is None:
                return head
            
            # Terminate if head reaches to the end
            if head.next is None:
                # Last node will be returned till the end of the recursion backward
                return head
            
            # Keep returning the end Node of the LinkedList
            node = reverse(head.next)
            
            # Reverse
            head.next.next = head
            head.next = None
            
            return node
            
        return reverse(head)
        
```
